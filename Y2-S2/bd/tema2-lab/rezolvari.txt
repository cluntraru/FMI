        LAST_NAME,
        FIRST_NAME
    FROM
        EMPLOYEES

        LAST_NAME IN
        (
            SELECT
                LAST_NAME
            FROM
            (
                SELECT

                    COUNT(LAST_NAME)

                    EMPLOYEES
                GROUP BY
                    LAST_NAME
                HAVING
                    COUNT(LAST_NAME) >= 2
            )
        )

6.
    SELECT
        DEPARTMENTS.DEPARTMENT_ID,
        DEPARTMENT_NAME
    FROM
        DEPARTMENTS
    JOIN
        (
            SELECT
                DEPARTMENT_ID,
                COUNT(DISTINCT JOB_ID)
            FROM
                EMPLOYEES
            GROUP BY
                DEPARTMENT_ID
            HAVING
                COUNT(DISTINCT JOB_ID) >= 2
        ) EMP
    ON
        DEPARTMENTS.DEPARTMENT_ID = EMP.DEPARTMENT_ID
